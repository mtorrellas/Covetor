=============================================
COMANDO SELECT
=============================================
SELECT        Gastos.ID, Gastos.Proveedor AS ProveedorID, Gastos.Fecha, Gastos.Factura, Gastos.Concepto, Gastos.TipoDestino, Gastos.Grupo AS GrupoID, Gastos.Finca AS FincaID, Gastos.Importe, 
                         Proveedores.RazonSocial, Grupos.Nombre AS Grupo, Fincas.Modelo AS ModeloID, Fincas.Identificacion AS Finca, Modelos.Modelo
FROM            Gastos LEFT OUTER JOIN
                         Proveedores ON Gastos.Proveedor = Proveedores.ID LEFT OUTER JOIN
                         Grupos ON Gastos.Grupo = Grupos.ID LEFT OUTER JOIN
                         Fincas ON Gastos.Finca = Fincas.ID LEFT OUTER JOIN
                         Modelos ON Fincas.Modelo = Modelos.Id
ORDER BY Gastos.Fecha DESC

COMANDO UPDATE

UPDATE       Gastos

WHERE        (ID = @Original_ID);
